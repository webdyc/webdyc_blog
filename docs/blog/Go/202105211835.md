## 定义
接口是一类规范，是某些方法的集合
## 使用
1. 写法一
```go
package main

import "fmt"

type Animal interface {
	Eat()
	Run()
}
type Cat struct {
	Name string
	sex bool
}
type Dog struct {
	Name string
}

func (c Cat)Run(){
	fmt.Println(c.Name,"开始跑")
}
func (c Cat)Eat()  {
	fmt.Println(c.Name,"开始吃")
}
func (d Dog)Run(){
	fmt.Println(d.Name,"开始跑")
}
func (d Dog)Eat()  {
	fmt.Println(d.Name,"开始吃")
}

func main()  {
	var a Animal
	c:=Cat{
		Name: "Tom",
		sex: false,
	}
	a = c
	a.Eat()
	a.Run()
}
```

	Eat()
	Run()
}

	Name string
	sex bool
}
type Dog struct {
	Name string
}

func (c Cat)Run(){
	fmt.Println(c.Name,"开始跑")

	fmt.Println(d.Name,"开始跑")

var L Animal
func main()  {
	c:=Cat{
