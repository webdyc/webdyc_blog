## 声明变量
- var 缺点
1. 污染全局
```js script  
    var a = 1
    console.log(window.a)
```
2. 导致变量提升
```js script  
    //function var都会导致提升
    console.log(a) 
    var a = 1
```
3. 可以被重复声明
```js script  
    var a = 1;
    var b = 2;
    var c = 3;
```
4. 作用域问题 (常见的作用域: 全局作用域、函数作用域)
```js script  
    {
        var a = 1
    }
    console.log(a)
```
- let 变量
```js script  
    let a = 100
    {
        console.log(a) //暂存死区
        let a = 200
    }
```
- const 常量
```js script  
    // const 常量 不能重新赋值 (地址不变即可)
    const PI = {r: '3.14'}
    PI.r = 3.15
```

## 解构赋值
- ... 展开运算符
```js script  
    // 把两个数组合并成一个数组
    let arr1 = [1,2,3]
    let arr2 = [4,5,6]
    arr1.concat(arr2)
    let arr3 = [...arr1,...arr2]
    console.log(arr3)
    // 把两个对象合并成一个对象
    let school = {name:'dyc'}
    let my = {age:18}
    let all = {...school,...my}
    console.log(all)
```

## set/map 是两种存储结构
- set 集合 不能重复的东西,会去重
```js script  
    let s = new Set([1,2,3,4,1,2,3,4])
    console.log(typeof s) // object
    // 添加和删除，没有顺序
    s.add('5')
    s.delete('5')
    s.entries()
    s.keys()
    let arr = [...s]
    console.log(arr)
```
- map 映射表 有key,不能放重复的 可以循环调用
```js script  
    let m = new Map()
    m.set('name','dyc')
    let obj = {name:1}
    m.set(obj,'426') //obj引用空间被set所引用
    obj = null //obj清空 这个空间还是在的
    console.log(m)
```

## ES6 类 ES5 构造函数
- 了解构造函数的属性
```js
function Animal(){
    // 属性 分为两种 实例上的属性 公有属性
    this.name = name
    this.arr = [1,2,3]
}
Animal.prototype.address = {location: '山里'}
let a1 = new Animal('猴子')
let a2 = new Animal('老虎')
console.log(a1.arr === a2.arr) //false
console.log(a1.address === a2.address) //true
// 每个实例都有一个_proto_指向所属类的原型
console.log(a1._proto_ === Animal.prototype) //true
console.log(a1.constructor === Animal) //true
console.log(Animal_proto_ === Function.prototype) //true
console.log(a1._proto_._proto_ === Object.prototype) //true
console.log(Object.prototype._proto_)
```

## prxoy
1. Object.defineProperty 不支持数组更新 push slice  
2. 希望数组变化就能更新视图
```js
function update(){
    console.log('更新视图')
}
let arr = [1,2,3]
// proxy可以监控到数组的变化和对象变化
// 数组变化会改变数组的内容 还会改变数组长度
let proxy = new Proxy(arr,{
  set(target,key,value){
      // 不要手动调用原数组,因为数组变化时,可能调用的是push方法 pop这个时候key会有问题
      if(key === 'length') return true
      update()
      return Reflect.set(target,key,value)
  },
  get(target,key){
      return Reflect.get(target,key)
  }
})
proxy.push(123)
// console.log(proxy[0])
```
